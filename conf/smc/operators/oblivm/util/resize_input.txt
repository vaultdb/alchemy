package $packageName;

#define CARDINALITY $card

//naive resizing
int$size[CARDINALITY] run@v@w(int$size[v] a, int$size[w] b) {
	int tupleCount;
	getArrElements(a, tupleCount);
	
	public int card = $card;
	secure int$size[$card] dst;
	
	bfor(card) (int i=0; i<card; i=i+1) {
		dst[i] = a[i];
	}

	setArrElements(dst, tupleCount);
	return dst;
}
